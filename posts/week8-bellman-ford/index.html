#include<iostream>
#include<vector>

using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int tc;
	cin >> tc;

	for (int i = 0; i < tc; i++) {
		bool has_cycle = false;
		int n, m, w;
		cin >> n >> m >> w;

		long long dist[501];
		vector<pair<int, int>> d[501];

		for (int i = 1; i <= m + w; i++) {
			int s, e, t;
			cin >> s >> e >> t;
			if (i <= m) {
				d[s].push_back({ e,t });
				d[e].push_back({ s,t });
			}
			else d[s].push_back({ e,-t });
		}

		fill_n(dist, n + 1, 0);

		for (int i = 1; i <= n; i++) {
			for (int here = 1; here <= n; here++) {
				for (auto& p : d[here]) {
					int there = p.first;
					int d = p.second;

					if (dist[here] + d < dist[there]) {
						if (i == n) has_cycle = true;
						else dist[there] = dist[here] + d;
					}
				}
			}
		}

		cout << ((has_cycle) ? "YES" : "NO") << "\n";
	}
}
