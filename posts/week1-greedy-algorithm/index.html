int main(){
	int n, cnt;
	int max=0;
	cin>>n;

	vector<int> rope(n,0);
	
	for(int i=0; i<n; i++){
		cin>>rope[i];
	}
	
	sort(rope.begin(), rope.end(), greater<int>());
	
	for(cnt=1; cnt<=n; cnt++){
		if(rope[cnt-1]*cnt>max) max=rope[cnt-1]*cnt;
	}
	
	cout<<max;
}
