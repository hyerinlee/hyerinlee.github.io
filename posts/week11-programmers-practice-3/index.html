#include <vector>

using namespace std;

long long solution(int n, vector<int> times) {
    long long answer = 0;
    long long left = 1;
    long long right = 1'000'000'000'000'000'000;
    
    while(left<right){
        long long mid = (left+right)/2;
        long long headcount=0;
        for(const auto& t : times){
            headcount+=(mid/t);
        }
        if(n<=headcount){
            answer=mid;
            right=mid;
        }
        else left=mid+1;
    }
    
    return answer;
}

#include <queue>
#include <vector>

using namespace std;

int solution(int n, vector<vector<int>> edge) {
    int answer = 0;
    int farthest = 0;
    vector<int> count;
    queue<int> q;

    count.assign(n+1,-1);
    count[1]=0;
    q.push(1);
    while(!q.empty()){
        int here = q.front();
        q.pop();

        for(const auto& e : edge){
            if(e[0]==here && count[e[1]]==-1){
                count[e[1]]=count[e[0]]+1;
                farthest=count[e[0]]+1;
                q.push(e[1]);
            }
            else if(e[1]==here && count[e[0]]==-1){
                count[e[0]]=count[e[1]]+1;
                farthest=count[e[1]]+1;
                q.push(e[0]);
            }
        }
    }

    for(const auto& c : count){
        if(c==farthest) answer++;
    }

    return answer;
}
